CREATE DATABASE webserver;
USE webserver; 
CREATE TABLE device (
   device_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
   sign VARCHAR(128)
) ENGINE=InnoDB CHARSET=utf8;

CREATE TABLE data (
   data_id INTEGER NOT NULL AUTO_INCREMENT PRIMARY KEY,
   date_post VARCHAR(30),
   nhietdo VARCHAR(30),
   doam VARCHAR(30)
) ENGINE=InnoDB CHARSET=utf8;
